8.10 영역 칠하기 (그림판의 paint 기능 구현)

```java
class Solver {
	int N, M, X, Y;
	int[][] A;
	int C; // color of initial point
	int NC = 0; //new color

	Solver() throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
		StringTokenizer st;

		st = new StringTokenizer(br.readLine());
		N = Integer.parseInt(st.nextToken());
		M = Integer.parseInt(st.nextToken());
		X = Integer.parseInt(st.nextToken());
		Y = Integer.parseInt(st.nextToken());

		A = new int[N][M];
		for (int i = 0; i < N; i++) {
			st = new StringTokenizer(br.readLine());
			for (int j = 0; j < M; j++) {
				A[i][j] = Integer.parseInt(st.nextToken());
			}
		}

		C = A[X][Y];
		A[X][Y] = NC;
		paint(X, Y);

		for (int i = 0; i < N; i++) {
			for (int j = 0; j < M; j++) {
				bw.write(A[i][j]+ " ");
			}
			bw.write("\n");
		}
		bw.flush();
		bw.close();
	}

	int[] dx = { -1, 0, 0, 1 };
	int[] dy = { 0, -1, 1, 0 };

	private void paint(int x, int y) {
		for(int i = 0; i < 4; i++){
			int nx = x + dx[i];
			int ny = y + dy[i];
			if(0 <= nx && nx < N && 0 <= ny && ny < M && C == A[nx][ny] && A[nx][ny] != NC){
				A[nx][ny] = NC;
				paint(nx, ny);
			}
		}
	}
}
```

8.13 박스쌓기: 크기가 w, h, d인 박스가 n개 있을때 w, h, d가 항상 작은걸 큰거위에 올려놓을수있다. 박스탑의 최대 높이는?
```java
class Box {
	int x, y, z;

	public Box(int x, int y, int z) {
		super();
		this.x = x;
		this.y = y;
		this.z = z;
	}

	@Override
	public String toString() {
		return "Box [x=" + x + ", y=" + y + ", z=" + z + "]";
	}

}

class Solver {
	int N;
	Box[] A;
	int[] D;

	Solver() throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
		StringTokenizer st;

		st = new StringTokenizer(br.readLine());
		N = Integer.parseInt(st.nextToken());

		A = new Box[N];
		D = new int[N];
		for (int i = 0; i < N; i++) {
			st = new StringTokenizer(br.readLine());
			int x = Integer.parseInt(st.nextToken());
			int y = Integer.parseInt(st.nextToken());
			int z = Integer.parseInt(st.nextToken());
			A[i] = new Box(x, y, z);
		}
		Arrays.sort(A, (a, b) -> {
			if (a.x == b.x) {
				if (a.y == b.y) {
					return b.z - a.z;
				}
				return a.y - b.y;
			}
			return a.x - b.x;
		});

		for (int i = 0; i < N; i++) {
			D[i] = A[i].z;
		}

		for (int i = 0; i < N; i++) {
			Box cb = A[i];
			for (int j = i + 1; j < N; j++) {
				Box nb = A[j];
				if (cb.x < nb.x && cb.y < nb.y && cb.z < nb.z) {
					D[j] = Math.max(D[j], D[i] + nb.z);
					break;
				}
			}
		}

		int max = -1;
		for (int i = 0; i < N; i++) {
//			System.out.println((i + 1) + " " + D[i]);
			max = Math.max(max, D[i]);
		}
		bw.write(max + "\n");
		bw.flush();
		bw.close();
	}
}
```
